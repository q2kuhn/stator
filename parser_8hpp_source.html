40  //These are left-associative operators, so we expect their RBP
41  //to be BP+1, and they can group with themselves
42  _operators["+"].reset(new BinaryOpToken<detail::Add, 20, 21, 20>());
43  _operators["-"].reset(new BinaryOpToken<detail::Subtract, 20, 21, 20>());
44  _operators["*"].reset(new BinaryOpToken<detail::Multiply, 30, 31, 30>());
45  _operators["/"].reset(new BinaryOpToken<detail::Divide, 30, 31, 30>());
46 
47  //Power is right-associative, thus its RBP is equal to its BP

75  if (next() != token)

80  void consume() {
81  _start = _end;
82 
83  //Skip whitespace
84  while ((_str[_start] == ' ') && (_start < _str.size())) ++_start;
85 
86  _end = _start;
87 
88  //Check for sequence end
89  if (_start == _str.size())
90  return;
91 
92  //Not at end of sequence so at least one character in symbol
93  _end = _start + 1;
94 
95  //Parse numbers with decimal points and (possible signed)
96  //exponents. Signs at the front of numbers are parsed as unary
97  //operators.
98  if (std::isdigit(_str[_start])) {
99  bool decimal = false;
100  bool exponent = false;
101 
102  while (_end != _str.size()) {
103  if (_str[_end] == '.') {
104  if (!decimal && !exponent) {
105  decimal = true;
106  ++_end;
107  continue;
108  } else {
109  stator_throw() << "Unexpected decimal point?\n" << parserLoc();
110  }
111  }
112 
113  if ((_str[_end] == 'e') || (_str[_end] == 'E')) {
114  if (!exponent) {
115  exponent = true;
116  decimal = true; //Don't allow decimals in the exponent
117  ++_end;
118 
119  if (_end == _str.size())
120  stator_throw() << "String ended during parsing of exponent\n" << parserLoc();
121 
122  //Eat the exponent sign if present
123  if ((_str[_end] == '+') || (_str[_end] == '-'))
124  ++_end;
125 
126  if (_end == _str.size())
127  stator_throw() << "String ended during parsing of exponent\n" << parserLoc();
128 
129  if (!std::isdigit(_str[_end]))
130  stator_throw() << "Malformed exponent?\n" << parserLoc();
131 
132  continue;
133  } else {
134  stator_throw() << "Malformed exponent?\n" << parserLoc();
135  }
136  }
137  if (std::isdigit(_str[_end])) {
138  ++_end;
139  continue;
140  }
141 
142  break;
143  }
144  return;
145  }
146 
147  //Parsing a string
148  if (std::isalpha(_str[_start])) {
149  while ((_end < _str.size()) && std::isalpha(_str[_end]))
150  ++_end;
151  return;
152  }
153 
154  //Allow non-alpha single character operators (longer operators should be above!)
155  if (_operators.find(_str.substr(_start, 1)) != _operators.end())
156  return;
157 
158  stator_throw() << "Unrecognised token \"" << _str.substr(_start) << "\"\n" << parserLoc();
159  }

161  std::string parserLoc() {
162  return _str + "\n"
163  + std::string(_start, ' ') + std::string(_end - _start -1, '-') + "^";
164  }

233  Expr P() {
234  std::string token = next();
235  consume();
236 
237  if (token.empty())
238  stator_throw() << "Unexpected end of expression?\n" << parserLoc();
239 
240  //Parse numbers
241  if (std::isdigit(token[0])) {
242  std::stringstream ss;
243  ss << token;
244  double val;
245  ss >> val;
246  return Expr(val);
247  }
248 
249  //Parse unary operators which are prefixes
250  auto it = _operators.find(token);
251  if (it != _operators.end()) {
252  if (!dynamic_pointer_cast<PrefixOp>(it->second))
253  stator_throw() << "Operator is not a prefix/unary operator?\n" << parserLoc();
254  return it->second->apply(Exp(it->second->BP()), *this);
255  }
256 
257  //Its not a prefix operator or a number, if it is a single
258  //alpha character, then assume its a variable!
259  if ((token.size() == 1) && (std::isalpha(token[0])))
260  return Expr(VarRT(token[0]));
261 
262  stator_throw() << "Could not parse as a valid token?\n" << parserLoc();
263  }

265  Expr Exp(int p) {
266  Expr t = P();
267 
268  int r = std::numeric_limits<int>::max();
269  while (true) {
270  std::string token = next();
271 
272  //Handle the special case of end of string
273  if (token.empty()) break;
274 
275  auto it = _operators.find(token);
276 
277  if (it == _operators.end())
278  stator_throw() << "Expected binary or postfix operator but got \""<< token << "\"?\n" << parserLoc();
279 
280  if (dynamic_pointer_cast<PrefixOp>(it->second))
281  stator_throw() << "Expected binary or postfix operator but found Prefix operator \""<< token << "\"?\n" << parserLoc();
282 
283  if ((p > it->second->BP()) || (it->second->BP() > r)) break;
284 
285  consume();
286  t = it->second->apply(t, *this);
287  r = it->second->NBP();
288  }
289 
290  return t;
291  }
