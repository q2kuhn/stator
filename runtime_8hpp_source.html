468  template<typename Op>
469  Expr apply(const BinaryOpRT<Op>& op) {
470  Expr l = op.getLHS()->visit(*this);
471  Expr r = op.getRHS()->visit(*this);
472 
473  if (r == typename Op::right_zero())
474  return typename Op::right_zero();
475 
476  if (l == typename Op::left_zero())

483  return r;
