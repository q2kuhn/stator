865  inline StackVector<double, 3> solve_real_roots(const Polynomial<3, double, PolyVar>& f_original) {

867  if (f_original[3] == 0)
868  return solve_real_roots(change_order<2>(f_original));
869 
870  if (f_original[0] == 0)
871  //If the constant is zero, one root is x=0. We can divide
872  //by x and solve the remaining quadratic
873  return deflate_and_solve_polynomial(f_original, 0.0);
874 
875  //Convert to a cubic with a unity high-order coefficient
876  auto f = expand(f_original / f_original[3]);
877 
878  if ((f[2] == 0) && (f[1] == 0))
879  //Special case where f(x) = x^3 + f[0]
880  return StackVector<double, 3>{std::cbrt(-f[0])};
881 
882  static const double maxSqrt = std::sqrt(std::numeric_limits<double>::max());
883 
884  if ((f[2] > maxSqrt) || (f[2] < -maxSqrt))
885  //The equation is limiting to x^3 + f[2] * x^2 == 0. Use
886  //this to estimate the location of one root, polish it up,
887  //then deflate the polynomial and solve the quadratic.
888  return deflate_and_solve_polynomial(f, -f[2]);
889 
890  //NOT SURE THESE RANGE TESTS ARE BENEFICIAL
891  if (f[1] > maxSqrt)
892  //Special case, if f[1] is large (and f[2] is not) the root is
893  //near -f[0] / f[1], the x^3 term is negligble, and all other terms
894  //cancel.
895  return deflate_and_solve_polynomial(f, -f[0] / f[1]);
896 
897  if (f[1] < -maxSqrt)
898  //Special case, equation is approximated as x^3 + q x == 0
899  return deflate_and_solve_polynomial(f, -std::sqrt(-f[1]));
900 
901  if ((f[0] > maxSqrt) || (f[0] < -maxSqrt))
902  //Another special case where equation is approximated as f(x)= x^3 +f[0]
903  return deflate_and_solve_polynomial(f, -std::cbrt(f[0]));
904 
905  const double v = f[0] + (2.0 * f[2] * f[2] / 9.0 - f[1]) * (f[2] / 3.0);
906 
907  if ((v > maxSqrt) || (v < -maxSqrt))
908  return deflate_and_solve_polynomial(f, -f[2]);
909 
910  const double uo3 = f[1] / 3.0 - f[2] * f[2] / 9.0;
911  const double u2o3 = uo3 + uo3;
912 
913  if ((u2o3 > maxSqrt) || (u2o3 < -maxSqrt))
914  {
915  if (f[2]==0)
916  {
917  if (f[1] > 0)
918  return deflate_and_solve_polynomial(f, -f[0] / f[1]);
919 
920  if (f[1] < 0)
921  return deflate_and_solve_polynomial(f, -std::sqrt(-f[1]));
922 
923  return deflate_and_solve_polynomial(f, 0.0);
924  }
925 
926  return deflate_and_solve_polynomial(f, -f[1] / f[2]);
927  }
928 
929  const double uo3sq4 = u2o3 * u2o3;
930  if (uo3sq4 > maxSqrt)
931  {
932  if (f[2] == 0)
933  {
934  if (f[1] > 0)
935  return deflate_and_solve_polynomial(f, -f[0] / f[1]);
936 
937  if (f[1] < 0)
938  return deflate_and_solve_polynomial(f, -std::sqrt(-f[1]));
939 
940  return deflate_and_solve_polynomial(f, 0.0);
941  }
942 
943  return deflate_and_solve_polynomial(f, -f[1] / f[2]);
944  }
945 
946  const double j = (uo3sq4 * uo3) + v * v;
947 
948  if (j > 0)
949  {//Only one root (but this test can be wrong due to a
950  //catastrophic cancellation in j) (i.e. (uo3sq4 * uo3) == v
951  //* v) so we solve for this root then solve the deflated
952  //quadratic.
953 
954  const double w = std::sqrt(j);
955  double root;
956  if (v < 0)
957  root = std::cbrt(0.5*(w-v)) - (uo3) * std::cbrt(2.0 / (w-v)) - f[2] / 3.0;
958  else
959  root = uo3 * std::cbrt(2.0 / (w+v)) - std::cbrt(0.5*(w+v)) - f[2] / 3.0;
960 
961  //We don't test if the polishing fails. Without established
962  //bounds polishing is hard to do for multiple roots, so we
963  //just accept a polished root if it is available
964  stator::numeric::halleys_method([&](double x){ return eval_derivatives<2>(f, x); }, root);
965 
966  return deflate_and_solve_polynomial(f, root);
967  }
968 
969  if (uo3 >= 0)
970  //Triple root detected
971  return StackVector<double, 3>{std::cbrt(v) - f[2] / 3.0};
972 
973  const double muo3 = - uo3;
974  double s = 0;
975  if (muo3 > 0)
976  {
977  s = std::sqrt(muo3);
978  if (f[2] > 0) s = -s;
979  }
980 
981  const double scube = s * muo3;
982  if (scube == 0)
983  return StackVector<double, 3>{ -f[2] / 3.0 };
984 
985  double t = - v / (scube + scube);
986  //Clamp t, as in certain edge cases it might numerically fall
987  //outside of the valid range of acos
988  t = std::min(std::max(t, -1.0), +1.0);
989  const double k = std::acos(t) / 3.0;
990  const double cosk = std::cos(k);
991 
992  StackVector<double, 3> roots{ (s + s) * cosk - f[2] / 3.0 };
993 
994  const double sinsqk = 1.0 - cosk * cosk;
995  if (sinsqk < 0)
996  return roots;
997 
998  double rt3sink = std::sqrt(3.0) * std::sqrt(sinsqk);
999  roots.push_back(s * (-cosk + rt3sink) - f[2] / 3.0);
1000  roots.push_back(s * (-cosk - rt3sink) - f[2] / 3.0);
1001 
1002  //We don't test if the polishing fails. Without established
1003  //bounds polishing is "hard" with multiple real roots, so we
1004  //just accept a polished root if it is available
1005  stator::numeric::halleys_method([&](double x){ return eval_derivatives<2>(f, x); }, roots[0]);
1006  stator::numeric::halleys_method([&](double x){ return eval_derivatives<2>(f, x); }, roots[1]);
1007  stator::numeric::halleys_method([&](double x){ return eval_derivatives<2>(f, x); }, roots[2]);
1008 
1009  std::sort(roots.begin(), roots.end());
1010  return roots;

1368  template<size_t Order, class Coeff_t, class PolyVar>
1369  StackVector<std::pair<Coeff_t,Coeff_t>, Order>
1370  VCA_real_root_bounds_worker(const Polynomial<Order, Coeff_t, PolyVar>& f) {
1371  //Test how many roots are in the range (0,1)
1372  switch (budan_01_test(f)) {
1373  case 0:
1374  //No roots, return empty
1375  return StackVector<std::pair<Coeff_t,Coeff_t>, Order>();
1376  case 1:
1377  //One root! the bound is (0,1)
1378  return StackVector<std::pair<Coeff_t,Coeff_t>, Order>{std::make_pair(Coeff_t(0), Coeff_t(1))};
1379  default:
1380  //Possibly multiple roots, so divide the range [0,1] in half
1381  //and recursively explore it.
1382 
1383  //We first scale the polynomial so that the original range
1384  //x=[0,1] is scaled to the range x=[0,2]. To do this, we
1385  //perform the subsitution x->x/2. Rather than use division
1386  //(even though a division by 2 usually cheap), we actually
1387  //generate this function \f$p1(x) = 2^{Order}\,f(x/2)\f$. This
1388  //transforms this to a multiplication op which is always cheap.
1389  Polynomial<Order, Coeff_t, PolyVar> p1(f);
1390  for (size_t i(0); i <= Order; ++i)
1391  p1[i] *= (1 << (Order-i)); //This gives (2^Order) / (2^i)
1392 
1393  //Perform a Taylor shift p2(x) = p1(x+1). This gives
1394  //
1395  //p2(x) = 2^Order f(x/2 + 0.5)
1396  //
1397  //in terms of the original function, f(x).
1398  const Polynomial<Order, Coeff_t, PolyVar> p2 = shift_function(p1, Unity());
1399 
1400  //Now that we have two scaled and shifted polynomials where
1401  //the roots in f in the range x=[0,0.5] are in p1 over the
1402  //range \f$x=[0,1]\f$, and the roots of f in the range
1403  //x=[0.5,1.0] are in p2 over the range \f$x=[0,1]\f$. Search
1404  //them both and combine the results.
1405

1407  //Scale these roots back to the original mapping
1408  for (auto& root_bound : retval) {
1409  root_bound.first /= 2;
1410  root_bound.second /= 2;
1411  }
1412 
1413  auto second_range = VCA_real_root_bounds_worker(p2);
1414  //Scale these roots back to the original mapping
1415  for (auto& root_bound : second_range)
1416  retval.push_back(std::make_pair(root_bound.first / 2 + 0.5, root_bound.second / 2 + 0.5));
1417 
1418  return retval;
1419  }
